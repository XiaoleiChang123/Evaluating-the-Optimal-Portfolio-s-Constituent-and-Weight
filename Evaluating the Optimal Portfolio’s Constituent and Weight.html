In [32]: def F_PortStd(cov, w_i):

    ...: portstd=[]

    ...: for i in range(w_i.shape[0]):

    ...: cov=np.array(cov)

    ...: w_i=np.array(w_i)

    ...: portvar=np.dot(w_i[i,:],np.dot(cov,w_i[i,:]))

    ...: port_std=math.sqrt(portvar)

    ...: portstd.append(port_std)

    ...: portstd = np.array(portstd)

    ...: return portstd
